# Funcție reutilizabilă pentru afișarea planurilor
async def show_plans(update_or_query, context):
    keyboard = [
        [InlineKeyboardButton(plan, callback_data=plan)]
        for plan in PLAN_DETAILS
    ]
    back_button = [[InlineKeyboardButton("🔙 Înapoi", callback_data="back_to_plans")]]
    reply_markup = InlineKeyboardMarkup(keyboard + back_button)

    # Dacă update_or_query este un obiect de tip Update (mesaj din chat)
    if isinstance(update_or_query, Update):
        await update_or_query.message.reply_text(
            "Alege planul dorit:",
            reply_markup=reply_markup
        )
    # Dacă update_or_query este un obiect de tip CallbackQuery (mesaj de la buton)
    elif isinstance(update_or_query, CallbackQuery):
        await update_or_query.answer()  # Răspunde la click
        await update_or_query.message.reply_text(
            "Alege planul dorit:",
            reply_markup=reply_markup
        )


# /start – mesaj de salut și afișarea planurilor
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await show_plans(update, context)

# Când se apasă pe un plan sau butonul „înapoi”
async def button_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = update.callback_query
    await query.answer()
    plan = query.data

    if plan == "back_to_plans":
        await show_plans(query, context)
        return

    details = PLAN_DETAILS[plan]

    # Trimite poza și descrierea planului + buton Înapoi
    back_button = [[InlineKeyboardButton("🔙 Înapoi", callback_data="back_to_plans")]]
    reply_markup = InlineKeyboardMarkup(back_button)

    await query.message.reply_photo(
        photo=details["image"],
        caption=f"{details['desc']}\n\n💳 Plătește aici:\n{details['link']}",
        reply_markup=reply_markup,
        parse_mode='Markdown'
    )

# Configurare aplicație
app = ApplicationBuilder().token(TOKEN).build()

app.add_handler(CommandHandler("start", start))
app.add_handler(CallbackQueryHandler(button_handler))

app.run_polling()
